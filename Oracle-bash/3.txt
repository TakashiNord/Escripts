#!/bin/bash

. /home/oracle/.bash_profile

justconnect(){
sqlplus -s -L  zabbix/"my_password"@my_database  << SQL
    set echo off;
    set tab off;
    set pagesize 0;
    set feedback off;
    set trimout on;
    set heading off;
    SELECT trunc(sum(bytes/1024/1024/1024),0) 
    FROM dba_free_space
    WHERE tablespace_name='MY_TABLESPACE';
SQL
}
RESULT=$( justconnect  );

#убираем пробелы
RESULT=`(echo ${RESULT} | fmt -su)`

#проверяем на число и точки
if [[ "${RESULT}"  =~ ^[0-9\.]+$ ]]; then
:
else
    RESULT="-1"
fi

echo $RESULT;
exit 0;
